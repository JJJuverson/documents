      # 员工页表

      <html>
  <body>
    <div class="wrap">
      <div class="ctrl">
        <div class="age_sort nu">
          <a href="javascrapt:;">年龄从小到大</a>
          <a href="javascrapt:;">年龄从大到小</a>
          <a href="javascrapt:;" class="active">默认顺序</a>
        </div>
        <div class="gender_show">
          <a href="javascrapt:;">只显示男性</a>
          <a href="javascrapt:;">只显示女性</a>
          <a href="javascrapt:;" class="active">默认</a>
        </div>
      </div>
      <table id="table">
        <!--<thead>
          <tr>
            <th>id</th>
            <th>姓名</th>
            <th>年龄</th>
            <th>性别</th>
          </tr>
         -->
        <tbody></tbody>
      </table>
    </div> 
    <script>
      let data=[
        {
          id:0,
          name:'小明',
          age:24,
          gender:'男'
        },
        {
          id:1,
          name:'小明',
          age:22,
          gender:'女'
        },
        {
          id:2,
          name:'小明',
          age:25,
          gender:'女'
        },
        {
          id:3,
          name:'小明',
          age:28,
          gender:'男'
        },
      ];
      {
        let ageEles = document.querySelectorAll(".age_sort a");
        let genderEles = document.querySelectorAll(".gender_show a");
        let sortIndex = 1;
        let genderIndex = 1;
        //视图渲染
        function render(data){
          let str = "",
          data.forEach((item,key)=>{
            str += `<tr>
              <th>${item.id}</th>
              <th>${item.name}</th>
              <th>${item.age}</th>
              <th>${item.gender}</th>
            </tr>`
          })
          document.querySelector("#table tbody").innerHTML = str;
        }
        //年龄排序统一管理
        let argSort = [
          ()=>{
            //年龄由小到大
            data.sort((r1,r2)=>r1.age-r2.age);
          },
          ()=>{
            //年龄由大到小
            data.sort((r1,r2)=>r2.age-r1.age);
          },
          ()=>{
            //通过id进行排列
            data.sort((r1,r2)=>r1.id-r2.id);
          },
        ]
        //性别筛选统一管理
        let filterType = [
          ()=>{//筛选男
            return data.filter(v=>v.gender==="男")
          },
          ()=>{//筛选女
            return data.filter(v=>v.gender==="女")
          },
          ()=>[...data] //不筛选
        ]
        
        ageEles.forEach((item,key)=>{
          item.onclick = function(){  //这里不能用到箭头函数，箭头函数的this指向了window
            //console.log(this);
            //把所有高亮去掉
            ageEles.forEach(v=>{
              v.classList.remove("active");
            })
            this.classList.add("active");
            argSort[key]();
            let newData = filterType[genderIndex]()
            render(newData);
            sortIndex = key;
          }
        })
        
        genderEles.forEach((item,key)=>{
          item.onclick = function(){
            genderEles.forEach(v=>{
              v.classList.remove("active");
            })
            this.classList.add("active");
            let newData = filterType[key]();
            argSort[sortIndex]();
            render(newData);
            genderIndex = key;
          }
        })
        
        
        
        render(data); 
      }
    
    
    
    </script>
    
    
    
    
  </body>
